import { Injectable } from '@angular/core';
import { ApiDataService } from './api-service';
import { mapToHttpParamsQuery, fmt } from '@vnas2-workspace/vnas2-core';
import { NewsDomainDTO } from '../models';

const router = {
  news_domain: `/api/news_domains`,
  getTree: `/api/news_domains/get_trees_with_parent_id`,
  news_domain_path: `/api/news_domains/{domainId}`
};

// name convension:
/**
 * load = GET
 * Add = Post
 * Update = Put
 * Remove = DElETE
 */

@Injectable()
export class NewsDomainApiService {
  constructor(private httpClient: ApiDataService) {}

  loadNewsDomain() {
    return this.httpClient.get(router.news_domain);
  }

  loadTrees(filter?: {
    page_number?: string;
    page_size?: string;
    parent_id?: string;
  }) {
    const params = mapToHttpParamsQuery(filter);
    return this.httpClient.get(router.getTree, params);
  }

  addNewsDomain(data?: NewsDomainDTO) {
    return this.httpClient.post(router.news_domain, data);
  }

  updateNewsDomain(domainId: string, data: NewsDomainDTO) {
    const uri = fmt(router.news_domain_path, { domainId });
    return this.httpClient.put(uri, data);
  }

  removeNewsDomain(domainId: string) {
    const uri = fmt(router.news_domain_path, { domainId });
    return this.httpClient.delete(uri);
  }
}
