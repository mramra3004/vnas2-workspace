import { Injectable } from '@angular/core';
import { ApiDataService } from './api-service';
import { mapToHttpParamsQuery, fmt } from '@vnas2-workspace/vnas2-core';
import { MonitorObjectDTO } from '../models';

const router = {
  monitor_objects: `/api/monitor_objects`,
  monitor_detail: `/api/monitor_objects/{monitorObjectId}`,
  create_monitor_object: `/api/monitor_objects`,
  remove_monitor_object: `/api/monitor_objects/{monitorObjectId}`,
  update_monitor_object: `/api/monitor_objects/{monitorObjectId}`,
  object_types: '/api/monitor_objects/object_types'
};

@Injectable()
export class MonitorObjectsService {
  constructor(private httpClient: ApiDataService) {}

  loadListObjectsAdvancedSearch(filter?: {
    page_number?: string;
    page_size?: string;
    object_name?: string;
    object_type: string;
  }) {
    const params = mapToHttpParamsQuery(filter);
    return this.httpClient.get(router.monitor_objects, params);
  }

  loadDetailMonitor(monitorObjectId: number) {
    const uri = fmt(router.monitor_detail, { monitorObjectId });
    return this.httpClient.get(uri);
  }

  createMonitorObject(data: MonitorObjectDTO) {
    return this.httpClient.post(router.create_monitor_object, data);
  }

  removeMonitorObject(monitorObjectId: number) {
    const uri = fmt(router.remove_monitor_object, { monitorObjectId });
    return this.httpClient.delete(uri);
  }

  updateMonitorObject(monitorObjectId: number, data: MonitorObjectDTO) {
    const uri = fmt(router.update_monitor_object, { monitorObjectId });
    return this.httpClient.put(uri, data);
  }

  loadObjectTypes() {
    return this.httpClient.get(router.object_types);
  }
}
