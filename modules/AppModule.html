import { BrowserModule } from '@angular/platform-browser';
import { BrowserAnimationsModule } from '@angular/platform-browser/animations';
import { NgModule } from '@angular/core';

import { AppRoutingModule } from './app-routing.module';
import { AppComponent } from './app.component';
import { LoggerModule } from '@vnas2/logger-lib';
import { environment } from '../environments/environment';
import { Vnas2ApiModule } from '@vnas2-workspace/vnas2-api';
import { Vnas2CoreModule } from '@vnas2-workspace/vnas2-core';
import { AppLoaderModule } from '@vnas2-workspace/vnas2-ui-commons';

@NgModule({
  declarations: [AppComponent],
  imports: [
    BrowserModule,
    BrowserAnimationsModule,
    AppRoutingModule,
    LoggerModule.forRoot({ enableDebug: environment.production }),
    AppLoaderModule,
    Vnas2CoreModule.forRoot({ enableProduction: environment.production }),
    Vnas2ApiModule.forRoot()
  ],
  providers: [],
  bootstrap: [AppComponent]
})
export class AppModule {}
