import { NgModule } from '@angular/core';
import { Routes, RouterModule } from '@angular/router';
import { SettingsPages } from './settings.global';
import { Vnas2SettingsLayoutComponent } from './layouts/vnas2-settings-layout/vnas2-settings-layout.component';
import { Vnas2SettingsNewsBasketComponent } from './pages/vnas2-settings-news-basket/vnas2-settings-news-basket.component';
import { Vnas2SettingsNewsDomainComponent } from './pages/vnas2-settings-news-domain/vnas2-settings-news-domain.component';
import { Vnas2SettingsSocialObjectComponent } from './pages/vnas2-settings-social-object/vnas2-settings-social-object.component';
import { Vnas2SettingsComponent } from './pages/vnas2-settings/vnas2-settings.component';
const routes: Routes = [
  {
    path: '',
    component: Vnas2SettingsLayoutComponent,
    children: [
      {
        path: 'organization',
        component: Vnas2SettingsComponent,
        data: { title: 'tổ chức', type: 'Organization' }
      },
      {
        path: 'person',
        component: Vnas2SettingsComponent,
        data: { title: 'đối tượng', type: 'Person' }
      },
      {
        path: 'country',
        component: Vnas2SettingsComponent,
        data: { title: 'quốc gia', type: 'Country' }
      },
      {
        path: 'domain',
        component: Vnas2SettingsNewsDomainComponent,
        data: { title: 'danh mục', type: 'NewsDomain' }
      },
      {
        path: 'news-basket',
        component: Vnas2SettingsNewsBasketComponent,
        data: { title: 'giỏ tin', type: 'NewsBacket' }
      },
      {
        path: 'social-object',
        component: Vnas2SettingsSocialObjectComponent,
        data: {
          title: 'đối tượng mạng xã hội',
          type: 'SocialObject'
        }
      },
      // default set router redirect path url prefix
      {
        path: '',
        redirectTo: 'organization',
        pathMatch: 'prefix'
      }
    ]
  }
];

@NgModule({
  imports: [RouterModule.forChild(routes)],
  exports: [RouterModule]
})
export class Vnas2SettingsRoutingModule {}
