import {
  Component,
  OnInit,
  Input,
  OnChanges,
  Output,
  EventEmitter,
  OnDestroy,
  ViewChildren,
  ElementRef,
  AfterViewChecked,
  Renderer2
} from '@angular/core';
import { NewsDomain } from '../../models/news-domain.model';
import { FlatTreeControl } from '@angular/cdk/tree';
import {
  MatTreeFlatDataSource,
  MatTreeFlattener,
  MatTreeNode
} from '@angular/material';
import { Subject } from 'rxjs';
interface TreeFlatNode {
  expandable: boolean;
  name: string;
  level: number;
  domain_id?: number;
}
export interface SelectNodeTree {
  domain_id: number;
  name: string;
  domain_table_name: string;
}

@Component({
  selector: 'app-tree-category-vertical',
  templateUrl: './tree-category-vertical.component.html',
  styleUrls: ['./tree-category-vertical.component.scss']
})
export class TreeCategoryVerticalComponent
  implements OnChanges, OnDestroy, AfterViewChecked {
  @Input() TreeData: NewsDomain[];
  @Output() selectedNodes: EventEmitter<any> = new EventEmitter();
  private selectNodeTree: { [key: string]: SelectNodeTree } = {};
  private dictionaryName: { [key: string]: string } = {};

  treeControl = new FlatTreeControl<TreeFlatNode>(
    node => node.level,
    node => node.expandable
  );

  treeFlattener = new MatTreeFlattener(
    this.transformer,
    node => node.level,
    node => node.expandable,
    node => node.childs
  );

  dataSource = new MatTreeFlatDataSource(this.treeControl, this.treeFlattener);

  @ViewChildren(MatTreeNode, { read: ElementRef }) treeNodes: ElementRef[];
  hasListener: any[] = [];
  oldHighlight: ElementRef;
  private classHighLight = 'background-highlight';

  constructor(private renderer: Renderer2) {}

  hasChild = (_: number, node: TreeFlatNode) => node.expandable;

  ngOnChanges() {
    if (this.TreeData) {
      this.dataSource.data = this.TreeData;
      this.dictionaryName = this.mapTreeToDictionary(
        this.TreeData,
        'domain_table_name',
        'domain_description',
        'childs'
      );
    }
  }

  ngOnDestroy() {
    this.selectNodeTree = {};
  }

  updateHighlight(newHighlight: ElementRef) {
    if (this.oldHighlight) {
      this.renderer.removeClass(
        this.oldHighlight.nativeElement,
        this.classHighLight
      );
    }
    this.renderer.addClass(newHighlight.nativeElement, this.classHighLight);
    this.oldHighlight = newHighlight;
  }

  ngAfterViewChecked() {
    this.treeNodes.forEach(reference => {
      if (!this.hasListener.includes(reference.nativeElement)) {
        // console.log('* tick');

        this.renderer.listen(reference.nativeElement, 'click', () => {
          this.updateHighlight(reference);
        });
        this.renderer.listen(
          reference.nativeElement.children.item(0),
          'click',
          () => {
            this.updateHighlight(reference);
          }
        );
        this.hasListener = this.hasListener.concat([reference.nativeElement]);
      }
    });

    this.hasListener = this.hasListener.filter(element =>
      document.contains(element)
    );
    // console.log('*', this.hasListener.length);
  }

  get getSelectNodeTree() {
    return this.selectNodeTree;
  }

  get getDictionaryName() {
    return this.dictionaryName;
  }

  setSelectNodeTree(node: { [key: string]: SelectNodeTree }) {
    this.selectNodeTree = node;
  }

  selectNode(event: any) {
    if (event && event.expandable) {
      // expand after select node
      this.treeControl.expand(event);
    }

    const { domain_id, domain_table_name, expandable, name } = event;

    let selectedNodes: SelectNodeTree[] = [];

    if (expandable) {
      const trees = this.searchTree(
        this.TreeData,
        'domain_id',
        'childs',
        domain_id
      );
      if (trees && trees.length > 0) {
        const nodeTree = trees[0];
        if (nodeTree.childs && nodeTree.childs.length === 0) {
          const node: SelectNodeTree = {
            domain_id: nodeTree.domain_id,
            domain_table_name: nodeTree.domain_table_name,
            name: nodeTree.domain_description
          };
          selectedNodes = [node];
        } else {
          selectedNodes = nodeTree.childs.map(item => {
            const node: SelectNodeTree = {
              domain_id: item.domain_id,
              domain_table_name: item.domain_table_name,
              name: item.domain_description
            };
            return node;
          });
        }
      }
    }

    selectedNodes.unshift({
      domain_id,
      domain_table_name,
      name
    });
    // map event node select -> return selectNodeTree[]
    // when click node is parrent
    // when click node is children
    const selectedNodeDictionary = this.mapArrayToDictionary(
      selectedNodes,
      'domain_table_name'
    );
    this.setSelectNodeTree(selectedNodeDictionary);
    setTimeout(() => {
      this.selectedNodes.emit({ data: selectedNodes });
    }, 200);
  }

  private transformer(node: NewsDomain, level: number) {
    return {
      expandable: !!node.childs && node.childs.length > 0,
      name: node.domain_description,
      level,
      domain_id: node.domain_id,
      domain_table_name: node.domain_table_name,
      parent_domain_id: node.parent_domain_id
    };
  }

  private mapArrayToDictionary<T>(data: T[], key: string) {
    const dataDictionary: { [key: string]: T } = {};
    data.forEach(item => {
      dataDictionary[item[key]] = item;
    });
    return dataDictionary;
  }

  private mapTreeToDictionary<T>(
    data: T[],
    key: string,
    prod: string,
    child?: string
  ) {
    const dataDictionary: { [key: string]: string } = {};
    try {
      data.forEach(item => {
        dataDictionary[item[key]] = item[prod];
        if (item[child] && item[child].length > 0) {
          item[child].forEach(childItem => {
            dataDictionary[childItem[key]] = childItem[prod];
          });
        }
      });
    } catch {
      console.log('error...');
    }
    return dataDictionary;
  }

  private searchTree<T>(
    elements: T[],
    prod: string,
    children: string,
    matchingTitle: string
  ) {
    let result: T[] = [];
    if (elements === undefined) {
      return [];
    }
    result = elements.filter(item => item[prod] === matchingTitle);
    if (result.length === 1) {
      return result;
    }
    result = this.searchTree(result[children], prod, children, matchingTitle);
    return result;
  }
}
