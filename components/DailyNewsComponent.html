import {
  Component,
  Input,
  Output,
  EventEmitter,
  OnChanges
} from '@angular/core';
import { DailyNew } from '../../models';
import { fadeInAnimation } from '../../animations';

@Component({
  selector: 'app-daily-news',
  templateUrl: './daily-news.component.html',
  styleUrls: ['./daily-news.component.scss'],
  animations: [fadeInAnimation]
})
export class DailyNewsComponent implements OnChanges {
  @Input() dailyNews: DailyNew[] = null;
  @Output() clickDailyNew: EventEmitter<any> = new EventEmitter<any>();
  @Output() sortOrderField: EventEmitter<{
    data: { [key: number]: boolean };
  }> = new EventEmitter();

  dataSource = {
    data: this.dailyNews
  };

  titleSort = false; // 0
  dateSort = false; // 1
  ratingSort = false; // 2

  constructor() {}

  ngOnChanges() {
    this.dataSource.data = null;
    if (this.dailyNews) {
      setTimeout(() => {
        this.dataSource = {
          data: this.dailyNews
        };
      }, 600);
    }
  }

  onClickDailyNew(data: DailyNew) {
    this.clickDailyNew.emit({ data });
  }

  sortOrderBy(sort?: number) {
    const orderBy: { [key: number]: boolean } = {};

    switch (sort) {
      case 0:
        orderBy[0] = this.titleSort;
        this.titleSort = !this.titleSort;
        break;
      case 1:
        orderBy[1] = this.dateSort;
        this.dateSort = !this.dateSort;
        break;
      case 2:
        orderBy[2] = this.ratingSort;
        this.ratingSort = !this.ratingSort;
        break;
      default:
        break;
    }
    this.sortOrderField.emit({ data: orderBy });
  }
}
